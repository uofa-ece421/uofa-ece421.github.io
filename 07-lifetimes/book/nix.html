Lifetimes

nix

nix provides a rust friendly wrapping over OS calls that are normally unsafe.

Examples from the nix::sys::socket module: 

#![allow(unused)]
fn main() {
pub fn socket<T: Into<Option<SockProtocol>>>(
    domain: AddressFamily,
    ty: SockType,
    flags: SockFlag,
    protocol: T
) -> Result<OwnedFd>


pub fn listen<F: AsFd>(sock: &F, backlog: Backlog) -> Result<()>

pub fn accept(sockfd: RawFd) -> Result<RawFd>

pub fn read(fd: RawFd, buf: &mut [u8]) -> Result<usize>

pub fn write<Fd: AsFd>(fd: Fd, buf: &[u8]) -> Result<usize>
}

This API is based on std::os::fd::OwnedFd which is a data structure meant to encapsulate the lifetime of a file desciptor. The underlying RawFd is "owned" when the file is opened, and closes the descriptor on drop. It also guards against double close() by not implementing Copy.

To use the OwnedFd you need to use traits:

use nix::sys::socket::{socket, AddressFamily, SockFlag, SockType};
use nix::unistd::{read, write};
use std::os::fd::{AsFd, AsRawFd, OwnedFd};

fn main() {
    let fd: OwnedFd = socket(AddressFamily::Unix,
                             SockType::SeqPacket,
                             SockFlag::empty(),
                             None).unwrap();

    let mut buf: [u8; 128] = [0; 128];

    read(fd.as_raw_fd(), &mut buf).unwrap();

    write(fd.as_fd(), &mut buf).unwrap();
}
